    <script>
        function sendConsulta() {
            fetch('/mype/send-consulta', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').getAttribute('content')
                }
            })
            .then(response => response.json())
            .then(data => {
                alert(data.message || 'Consulta enviada');
                if (data.success) {
                    location.reload();
                }
            })
            .catch(error => {
                alert('Error: ' + error.message);
            });
        }

        function getStats() {
            fetch('/mype/stats')
            .then(response => response.json())
            .then(data => {
                alert(data.output || 'Estad√≠sticas obtenidas');
            })
            .catch(error => {
                alert('Error: ' + error.message);
            });
        }

        function resetQuestions() {

                    if (data.success) {

                    }
                })
                .catch(error => {
                    alert('Error: ' + error.message);
                });
            }
        }
